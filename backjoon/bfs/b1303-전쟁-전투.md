import java.util.*;
import java.lang.*;
import java.io.*;

// The main method must be in a class named "Main".
class Main {
    static int n;
    static int m;
    static int wCount = 0;
	static int bCount = 0;
    static int dy[] = {-1, 1, 0, 0}; 
	static int dx[] = {0, 0, -1, 1}; 
	static boolean visited[][];
	static char map[][];
    
    public static void main(String[] args) throws IOException {
        BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
        StringTokenizer st = new StringTokenizer(br.readLine());
        n = Integer.parseInt(st.nextToken());
        m = Integer.parseInt(st.nextToken());
        map = new char[m][n];
        visited = new boolean[m][n];

        for(int i = 0; i < m; i++) {
            st = new StringTokenizer(br.readLine());
			String temp = st.nextToken();
            for(int j = 0; j < n; j++) {
                char ch = temp.charAt(j);
                map[i][j] = ch;
            }
        }

        for(int i=0; i<m; i++) {
			for(int j=0; j<n; j++) {
				if(!visited[i][j]) {
					if(map[i][j] == 'W') {
						int num = bfs(i, j, 'W');
						wCount += num * num;
					}
                    else {
						int num = bfs(i, j, 'B');
						bCount += num * num;
					}
				}
			}
		}

        System.out.println(wCount + " " + bCount);
    }

  public static int bfs(int x, int y, char ch) {
    Queue<Pair> qu = new LinkedList<>();
    qu.add(new Pair(x, y));
    visited[x][y] = true;  // Mark as visited when enqueued
    int count = 1;  // Start count at 1 to include the starting node
    
    while (!qu.isEmpty()) {
        Pair p = qu.poll();

        for (int i = 0; i < 4; i++) {
            int nx = p.x + dx[i];
            int ny = p.y + dy[i];

            if (nx < 0 || ny < 0 || nx >= m || ny >= n || visited[nx][ny] || ch != map[nx][ny]) {
                continue;
            }

            qu.add(new Pair(nx, ny));
            visited[nx][ny] = true;
            count++;
        }
    }
    return count;
}
}

class Pair{
    int x;
    int y;
    Pair(int x, int y){
        this.x = x;
        this.y = y;
    }
}
